                        <span className="text-xl">💰</span>
                        <span className="font-bold">{scores.economic}</span>
                    </div>
                    <div className="flex flex-col items-center text-emerald-600">
                        <span className="text-xl">📜</span>
                        <span className="font-bold">{scores.cultural}</span>

                    <div className="flex flex-col items-center text-rose-600">
                        <span className="text-xl">🛡️</span>
                        <span className="font-bold">{scores.social}</span>

                            <div className="bg-amber-500 h-2.5 rounded-full transition-all duration-300" style={{ width: `${getPercent()}%` }}></div>
                        </div>
                    </div>
                </div>
            );
        };

        const CapitalBarChart = ({ scores }) => {
            const maxEco = 33, maxCul = 36, maxSoc = 40;
            const getPercent = (val, max) => Math.min(100, Math.round((val / max) * 100));

                    <div>
                        <div className="flex justify-between mb-2 items-center"><span className="font-bold text-amber-700 flex items-center text-lg"><span className="text-2xl mr-3">💰</span> 物資補給箱</span><span className="font-bold text-amber-700 text-lg">{scores.economic} <span className="text-sm opacity-50">/ {maxEco}</span></span></div>
                        <div className="w-full bg-amber-100 rounded-full h-5 shadow-inner"><div className="bg-amber-500 h-5 rounded-full transition-all duration-1000 shadow-sm" style={{ width: `${getPercent(scores.economic, maxEco)}%` }}></div></div>

                    <div>
                        <div className="flex justify-between mb-2 items-center"><span className="font-bold text-emerald-700 flex items-center text-lg"><span className="text-2xl mr-3">📜</span> 天賦技能樹</span><span className="font-bold text-emerald-700 text-lg">{scores.cultural} <span className="text-sm opacity-50">/ {maxCul}</span></span></div>

                    <div>
                        <div className="flex justify-between mb-2 items-center"><span className="font-bold text-rose-700 flex items-center text-lg"><span className="text-2xl mr-3">🛡️</span> 盟友守護陣</span><span className="font-bold text-rose-700 text-lg">{scores.social} <span className="text-sm opacity-50">/ {maxSoc}</span></span></div>
                        <div className="w-full bg-rose-100 rounded-full h-5 shadow-inner"><div className="bg-rose-500 h-5 rounded-full transition-all duration-1000 shadow-sm" style={{ width: `${getPercent(scores.social, maxSoc)}%` }}></div></div>
                    </div>
                </div>
            );
        };

        // --- Main App ---
        const App = () => {
            const [gameState, setGameState] = useState('start'); 
            const [nickname, setNickname] = useState('');
            const [currentQIndex, setCurrentQIndex] = useState(0);
            const [scores, setScores] = useState({ economic: 0, cultural: 0, social: 0 });
            const [inventory, setInventory] = useState([]); 
            const [answers, setAnswers] = useState({});
            const [lastAnswer, setLastAnswer] = useState(null); 
            
            const scrollContainerRef = useRef(null);
            const activeTierRef = useRef(null);
            const resultRef = useRef(null);

            // Safer Access
            const currentQ = questionsData[currentQIndex] || questionsData[0];
            
            useEffect(() => {
                if (gameState === 'intro') {
                    const timer = setTimeout(() => setGameState('question'), 2500); 
                    return () => clearTimeout(timer);
                }
            }, [gameState]);

            useEffect(() => {
                if (gameState === 'result' && activeTierRef.current && scrollContainerRef.current) {
                    const container = scrollContainerRef.current;
                    const activeCard = activeTierRef.current;
                    const scrollLeft = activeCard.offsetLeft + (activeCard.offsetWidth / 2) - (container.offsetWidth / 2);
                    container.scrollTo({ left: scrollLeft, behavior: 'smooth' });
                }
            }, [gameState]);

            const calculateTier = (total) => {
                if (total >= 101) return { title: "傳說的守護者 (Legendary Guardian)", mode: "👑 創世無敵模式 (God / Creative Mode)", status: "哇！你幾乎擁有全套的神級裝備與強大的公會保護。你在遊戲中就像開了外掛，擁有改變規則的能力，世界對你來說充滿了無限可能。", advice: "擁有這樣的力量是一種巨大的責任。請小心不要因為過得太順遂，而誤以為別人的失敗是因為「不夠努力」。你的任務不僅僅是過關，而是「改變遊戲規則」。你有能力讓這個世界變得對所有人都更公平、更友善。請成為那位守護弱者的真正的英雄。" };
                if (total >= 76) return { title: "幸運的領航員 (Lucky Navigator)", mode: "🍀 幸運加成模式 (Lucky Mode)", status: "你的運氣很好，背包裡裝了許多稀有道具，這讓你比別人少走很多彎路，視野也更開闊。你可以更專注在探索未知的地圖。", advice: "恭喜你抽到了很好的開局手牌！請記得，這些優勢大多是運氣帶來的，而非單純是你個人的功勞。請保持謙虛與感恩。試著成為隊伍中的領航員，利用你的視野與資源，主動移除路上的障礙，讓隊友們也能順利前進。" };
                if (total >= 51) return { title: "踏實的冒險家 (Grounded Adventurer)", mode: "⚖️ 標準平衡模式 (Normal Mode)", status: "你的裝備平衡且實用，雖沒有傳說級的神裝，但足以應付大多數的關卡。你擁有穩定的力量，可以按部就班地完成目標。", advice: "你是這個社會的中堅力量。你的背包重量適中，這讓你既能理解資源匱乏者的辛苦，也能想像資源豐富者的生活。善用這份「中間視角」，成為連結不同群體的橋樑。你的穩定與友善，能讓這個隊伍走得更遠。" };
                if (total >= 26) return { title: "堅毅的挑戰者 (Resilient Challenger)", mode: "⚔️ 困難挑戰模式 (Hard Mode)", status: "你的背包裡有一些基礎道具，但路途依然崎嶇，常常需要為了資源而精打細算。你過得並不輕鬆，但也因為這樣，你更懂得珍惜每一分收穫。", advice: "你是一個腳踏實地的玩家。你知道什麼是「努力就有收穫」的感覺，但也體會過資源不足的無奈。這種同理心是你最大的優勢。請繼續保持你的勤奮，同時多觀察周遭的機會，你的每一次升級都是紮紮實實的實力證明。" };
                return { title: "逆境中的勇者 (Hero in Adversity)", mode: "🔥 地獄極限模式 (Extreme Survival Mode)", status: "你的背包非常沉重，裡面可能只有最基本的生存物資，甚至還背負著照顧他人的責任。這讓你必須比別人更努力、更小心才能前進。", advice: "這一路走來真的不容易，請為你還站在這裡感到驕傲。你正在玩這場遊戲最困難的模式，這鍛鍊出你超乎常人的韌性與生存智慧。這些是金錢買不到的寶藏。請相信，你的起點不決定你的終點，尋找像老師這樣值得信任的盟友，你絕對有能力走出屬於自己的傳奇路徑。" };
            };

            const totalScore = scores.economic + scores.cultural + scores.social;
            const tierInfo = calculateTier(totalScore);

            // Auto Save to Google Sheet
            useEffect(() => {
                if (gameState === 'result') {
                    const dataToSave = { 
                        nickname, 
                        totalScore, 
                        scores, 
                        tierTitle: tierInfo.title, 
                        tierMode: tierInfo.mode, 
                        answers: answers 
                    };
                    if (window.submitToGoogleSheet) { 
                        window.submitToGoogleSheet(dataToSave); 
                    }
                }
            }, [gameState]);

            const handleStart = (e) => {
                if (e) e.preventDefault();
                if (!nickname.trim()) return alert("請輸入冒險者暱稱！");
                setCurrentQIndex(0); 
                setGameState('instruction'); 
            };

            const handleInstructionAck = () => {
                setGameState('intro');
            };

            const handleAnswer = (isYes) => {
                if (isYes) {
                    setScores(prev => ({ ...prev, [currentQ.type]: prev[currentQ.type] + currentQ.weight }));
                    setInventory(prev => [...prev, { text: currentQ.text, type: currentQ.type }]);
                }
                setAnswers(prev => ({ ...prev, [currentQ.id]: isYes }));
                setLastAnswer(isYes);
                setGameState('feedback');
            };

            const handleUndo = () => {
                if (lastAnswer === true) {
                     setScores(prev => ({ ...prev, [currentQ.type]: prev[currentQ.type] - currentQ.weight }));
                    setInventory(prev => prev.slice(0, -1));
                }
                const newAnswers = { ...answers };
                delete newAnswers[currentQ.id];
                setAnswers(newAnswers);
                setGameState('question');
            };

            const handleNext = () => {
                if (currentQIndex < questionsData.length - 1) {
                    const nextIndex = currentQIndex + 1;
                    const currentLevel = questionsData[currentQIndex].level;
                    const nextLevel = questionsData[nextIndex].level;
                    setCurrentQIndex(nextIndex);
                    setGameState(nextLevel !== currentLevel ? 'intro' : 'question');
                } else {
                    setGameState('result');
                }
            };

            const handleRestart = () => {
                setScores({ economic: 0, cultural: 0, social: 0 });
                setInventory([]);
                setAnswers({});
                setCurrentQIndex(0);
                setGameState('start');
                setNickname('');
            };

            const handleScreenshot = async () => {
                if (!resultRef.current) return;
                try {
                    const canvas = await html2canvas(resultRef.current, { scale: 2, backgroundColor: '#ffffff' });
                    const image = canvas.toDataURL("image/png");
                    const link = document.createElement("a");
                    link.href = image;
                    link.download = `冒險證書_${nickname}.png`;
                    link.click();
                } catch (e) {
                    console.error("Screenshot failed", e);
                    alert("截圖失敗，請手動截圖保存。");
                }
            };

            const allTiers = [
                { min: 0, max: 25, label: "0 ~ 25分", icon: "🏅", title: "逆境中的勇者", mode: "地獄極限模式" },

                { min: 51, max: 75, label: "51 ~ 75分", icon: "🥈", title: "踏實的冒險家", mode: "標準平衡模式" },
                { min: 76, max: 100, label: "76 ~ 100分", icon: "🍀", title: "幸運的領航員", mode: "幸運加成模式" },

            ];

            const maxScore = 109;
            const percentage = Math.min(100, Math.round((totalScore / maxScore) * 100));
            const radius = 60;
            const circumference = 2 * Math.PI * radius;
            const strokeDashoffset = circumference - (percentage / 100) * circumference;

            // Render Views Logic
            return (
                <div className="min-h-screen flex flex-col">
                    {/* Header is always visible except start and instruction */}
                    {gameState !== 'start' && gameState !== 'instruction' && <Header scores={scores} currentQIndex={currentQIndex} totalQ={questionsData.length} />}
                    
                    <div className="flex-1 flex flex-col items-center justify-center p-4 max-w-2xl mx-auto w-full">
                        
                        {gameState === 'start' && (
                            <div className="min-h-screen flex flex-col items-center justify-center p-4 text-center w-full">
                                <div className="mb-8 text-6xl animate-bounce">🎒</div>
                                <h1 className="text-3xl font-bold text-amber-800 mb-2">探索你的隱形背包</h1>
                                <p className="text-amber-700 mb-8">社會學 RPG 冒險</p>
                                <div className="bg-white p-6 rounded-2xl shadow-xl w-full max-w-md mb-8">
                                    <form onSubmit={handleStart}>
                                        <label className="block text-left text-gray-600 mb-2 font-bold">冒險者暱稱</label>
                                        <input type="text" className="w-full p-3 border-2 border-amber-200 rounded-lg mb-4 text-lg" placeholder="請輸入你的名字..." value={nickname} onChange={(e) => setNickname(e.target.value)} />
                                        <button type="submit" className="w-full bg-amber-500 hover:bg-amber-600 text-white font-bold py-3 px-6 rounded-lg text-xl shadow-md transition transform hover:scale-105">🎒 背上背包，開始冒險</button>
                                    </form>
                                </div>
                            </div>
                        )}

                        {gameState === 'instruction' && (
                            <div className="min-h-screen flex flex-col items-center justify-center p-6 text-center fade-in w-full bg-amber-50 fixed inset-0 z-50">
                                <div className="max-w-lg bg-white p-8 rounded-3xl shadow-xl border-4 border-amber-200">
                                    <div className="text-6xl mb-6">⏳</div>
                                    <h2 className="text-2xl font-bold text-amber-800 mb-6 leading-relaxed">
                                        冒險者 <span className="text-amber-600">{nickname}</span> 您好！
                                    </h2>
                                    <p className="text-xl text-gray-700 mb-4 font-bold">
                                        請以你目前 <span className="text-rose-500 text-2xl">12歲</span> 👦👧<br/>的生命經驗進行回答......
                                    </p>
                                    <p className="text-gray-500 mb-8 text-sm">
                                        (如果你已經長大了，請試著乘坐時光機回到過去 🔙)
                                    </p>
                                    <button 
                                        onClick={handleInstructionAck}
                                        className="w-full bg-emerald-500 hover:bg-emerald-600 text-white font-bold py-4 px-6 rounded-xl text-xl shadow-lg transition transform hover:scale-105"
                                    >
                                        👌 沒問題，出發！
                                    </button>
                                </div>

                                <div className="text-8xl mb-6 emoji-animate">{currentQ.levelIcon}</div>

                                <h1 className="text-4xl font-bold tracking-wider">{currentQ.levelName}</h1>
                            </div>
                        )}

                        {gameState === 'question' && (
                            <div className="w-full fade-in">
                                <div className="bg-white rounded-3xl shadow-xl p-8 mb-6 relative overflow-hidden min-h-[300px] flex flex-col justify-center items-center text-center border-b-8 border-amber-200">
                                    <div className="text-sm font-bold text-amber-600 mb-4 tracking-widest uppercase bg-amber-100 px-4 py-1 rounded-full">Q{currentQIndex + 1} / {questionsData.length}</div>
                                    <span className="text-6xl mb-6 block">{currentQ.levelIcon}</span>
                                    <h2 className="text-xl md:text-2xl font-bold text-gray-800 leading-relaxed">{currentQ.text}</h2>
                                </div>
                                <div className="grid grid-cols-2 gap-4">
                                    <button onClick={() => handleAnswer(false)} className="bg-rose-50 hover:bg-rose-100 text-rose-700 border-2 border-rose-200 font-bold py-6 rounded-2xl text-lg md:text-xl transition transform active:scale-95 flex flex-col items-center shadow-sm"><span className="text-3xl mb-1">❌</span> 無體驗過</button>
                                    <button onClick={() => handleAnswer(true)} className="bg-emerald-50 hover:bg-emerald-100 text-emerald-700 border-2 border-emerald-200 font-bold py-6 rounded-2xl text-lg md:text-xl transition transform active:scale-95 flex flex-col items-center shadow-sm"><span className="text-3xl mb-1">⭕</span> 有體驗過</button>
                                </div>
                            </div>
                        )}

                        {gameState === 'feedback' && (
                            <div className="w-full fade-in">
                                <div className={`rounded-3xl shadow-xl p-8 mb-6 border-l-8 min-h-[300px] flex flex-col justify-center items-center text-center ${lastAnswer ? 'bg-amber-50 border-amber-400' : 'bg-slate-50 border-slate-400'}`}>
                                    <div className="mb-6">{lastAnswer ? <div className="inline-block bg-amber-200 text-amber-900 px-4 py-2 rounded-full font-bold text-lg shadow-sm">🎉 獲得裝備！</div> : <div className="inline-block bg-slate-200 text-slate-800 px-4 py-2 rounded-full font-bold text-lg shadow-sm">🛡️ 冒險者筆記</div>}</div>
                                    <p className="text-gray-700 text-lg md:text-xl leading-relaxed font-medium">{lastAnswer ? currentQ.yesText : currentQ.noText}</p>

                                    <button onClick={handleUndo} className="bg-gray-400 hover:bg-gray-500 text-white font-bold py-4 px-6 rounded-full shadow-lg text-lg flex items-center justify-center transition hover:scale-105">↩️ 重選</button>
                                    <button onClick={handleNext} className="bg-blue-600 hover:bg-blue-700 text-white font-bold py-4 px-10 rounded-full shadow-lg text-xl flex items-center justify-center transition hover:scale-105">下一題 ➡️</button>
                                </div>
                            </div>
                        )}

                        {gameState === 'result' && (
                            <div className="w-full">
                                <div className="bg-amber-600 text-white p-8 rounded-b-3xl shadow-lg mb-8 text-center -mx-4">
                                    <h1 className="text-2xl md:text-3xl font-bold mb-2">冒險結束！「{nickname}」的稱號是...</h1>
                                </div>
                                <div className="max-w-4xl mx-auto px-4" ref={resultRef}>
                                    <div className="bg-white rounded-2xl shadow-xl p-8 mb-8 transform -translate-y-6">
                                        <div className="flex flex-col md:flex-row items-center gap-8">
                                            <div className="flex-shrink-0 w-full md:w-auto flex flex-col items-center justify-center border-b md:border-b-0 md:border-r border-gray-100 pb-6 md:pb-0 md:pr-10">
                                                <h3 className="text-gray-400 font-bold mb-4 text-sm uppercase tracking-widest">你的總積分</h3>
                                                <div className="relative w-48 h-48 flex items-center justify-center">
                                                    <svg className="w-full h-full transform -rotate-90">
                                                        <circle cx="50%" cy="50%" r={radius} stroke="#f3f4f6" strokeWidth="12" fill="none" />
                                                        <circle cx="50%" cy="50%" r={radius} stroke="#f59e0b" strokeWidth="12" fill="none" strokeDasharray={circumference} strokeDashoffset={strokeDashoffset} strokeLinecap="round" style={{ transition: "stroke-dashoffset 1s ease-out" }} />
                                                    </svg>
                                                    <div className="absolute flex flex-col items-center">
                                                        <div className="text-6xl font-black text-amber-600 leading-none">{totalScore}</div>
                                                        <div className="text-xs text-gray-400 font-bold mt-1">/ {maxScore}</div>
                                                    </div>
                                                </div>
                                            </div>
                                            <div className="flex-grow space-y-6 w-full text-center md:text-left">
                                                <div className="border-b border-amber-100 pb-4">
                                                    <div className="text-gray-500 font-bold mb-1 text-sm">冒險者：<span className="text-gray-800 text-lg">{nickname}</span></div>
                                                    <h2 className="text-3xl font-bold text-gray-800 mb-2">{tierInfo.title}</h2>
                                                    <div className="inline-block bg-amber-100 text-amber-800 px-3 py-1 rounded-lg font-bold text-sm">{tierInfo.mode}</div>
                                                </div>
                                                <div className="space-y-4 text-gray-700 leading-relaxed text-left">
                                                    <div className="bg-slate-50 p-4 rounded-xl border-l-4 border-slate-400">
                                                        <h4 className="font-bold text-slate-700 mb-1 flex items-center">📊 狀態分析</h4>
                                                        <p className="text-sm md:text-base">{tierInfo.status}</p>
                                                    </div>
                                                    <div className="bg-amber-50 p-4 rounded-xl border-l-4 border-amber-400">
                                                        <h4 className="font-bold text-amber-700 mb-1 flex items-center">💡 給冒險者的建議</h4>
                                                        <p className="text-sm md:text-base">{tierInfo.advice}</p>
                                                    </div>
                                                </div>
                                            </div>
                                        </div>
                                    </div>
                                    <div className="bg-white rounded-2xl shadow-lg p-6 mb-6">
                                        <h3 className="text-lg font-bold text-gray-700 border-b pb-2 mb-4">📜 冒險者公會等級一覽表</h3>
                                        <div ref={scrollContainerRef} className="flex overflow-x-auto gap-4 pb-4 px-2 snap-x hide-scrollbar">
                                            {allTiers.map((tier, idx) => {
                                                const isActive = totalScore >= tier.min && totalScore <= tier.max;
                                                return (
                                                    <div key={idx} ref={isActive ? activeTierRef : null} className={`min-w-[260px] md:min-w-[200px] flex-1 flex flex-col justify-between p-4 rounded-xl border snap-center transition-all duration-300 ${isActive ? "bg-amber-100 border-amber-500 shadow-md scale-100 opacity-100 ring-2 ring-amber-300" : "bg-gray-50 border-gray-200 opacity-70 grayscale-[0.3]"}`}>
                                                        <div className="text-center mb-3">
                                                            <div className="text-4xl mb-2">{tier.icon}</div>
                                                            <div className={`font-bold text-lg leading-tight mb-1 ${isActive ? "text-amber-900" : "text-gray-600"}`}>{tier.title}</div>
                                                            <div className="text-xs text-gray-500">{tier.mode}</div>
                                                        </div>
                                                        <div className="text-center border-t border-black/10 pt-3">
                                                            <div className="text-sm font-mono font-bold text-gray-600 bg-white/50 inline-block px-2 rounded">{tier.label}</div>
                                                            {isActive && <div className="mt-2 text-xs font-bold text-amber-600 bg-white px-2 py-1 rounded-full shadow-sm inline-block">👈 你在這裡</div>}
                                                        </div>
                                                    </div>
                                                );
                                            })}
                                        </div>
                                    </div>
                                    <div className="bg-white rounded-2xl shadow-lg p-6 mb-6">
                                        <h3 className="text-lg font-bold text-gray-700 border-b pb-2 mb-4">📊 三大資本能力值</h3>
                                        <CapitalBarChart scores={scores} />
                                    </div>
                                    <div className="bg-white rounded-2xl shadow-lg p-6 mb-6">
                                        <h3 className="text-lg font-bold text-gray-700 border-b pb-2 mb-4">🎒 你的背包清單 ({inventory.length} 項)</h3>
                                        {inventory.length === 0 ? <p className="text-gray-500 text-center py-4">你的背包非常輕盈...</p> : (
                                            <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                                                {['economic', 'cultural', 'social'].map(type => {
                                                    const items = inventory.filter(i => i.type === type);
                                                    if (items.length === 0) return null;
                                                    let icon="", title="", color="";
                                                    if(type==='economic') { icon="💰"; title="物資補給箱"; color="bg-amber-50 border-amber-200 text-amber-800"; }
                                                    if(type==='cultural') { icon="📜"; title="天賦技能樹"; color="bg-emerald-50 border-emerald-200 text-emerald-800"; }
                                                    if(type==='social') { icon="🛡️"; title="盟友守護陣"; color="bg-rose-50 border-rose-200 text-rose-800"; }
                                                    return (
                                                        <div key={type} className={`rounded-xl p-4 border ${color} md:col-span-2`}>
                                                            <h4 className="font-bold flex items-center mb-3 text-lg border-b border-black/10 pb-2"><span className="mr-2 text-2xl">{icon}</span> {title}</h4>
                                                            <ul className="space-y-2">{items.map((item, idx) => <li key={idx} className="flex items-start"><span className="mr-2 opacity-60">•</span><span>{item.text}</span></li>)}</ul>
                                                        </div>
                                                    );
                                                })}
                                            </div>
                                        )}
                                    </div>
                                </div>
                                <div className="max-w-4xl mx-auto px-4 pb-8 flex flex-col gap-4">
                                    <button onClick={handleScreenshot} className="w-full bg-emerald-600 hover:bg-emerald-700 text-white font-bold py-4 rounded-xl shadow-lg transition transform hover:scale-[1.01]">📸 下載冒險證書</button>
                                    <button onClick={handleRestart} className="w-full bg-gray-800 hover:bg-black text-white font-bold py-4 rounded-xl shadow-lg transition transform hover:scale-[1.01]">🔄 再次挑戰</button>
                                </div>
                            </div>
                        )}
                    </div>
                </div>
            );
        };

        const root = ReactDOM.createRoot(document.getElementById('root'));
        root.render(<App />);
    </script>
